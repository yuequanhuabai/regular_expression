$/ =".\n";
while(<>){
    next if !s/\b([a-z]+)((?:\s/<[^>]+>)+)(\1\b)/\e[7m$1\e[m$2\e[7m$3\e[m/ig;]]]]
    s/^(?:[^\e]*\n)+      //mg;  #去除未标记的行
    s/^/$ARGV:   /mg;     #  在行首添加文件名
    print;
}

该程序的主要功能依靠3个正则表达式：
--  \b([a-z]+)((?:\s|<[^>]+>)+)(\1\b)
--  ^(?:[^\e]*\n)+
--  ^
